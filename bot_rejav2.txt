import google.generativeai as genai
from google.cloud.firestore_v1.base_vector_query import DistanceMeasure
from firebase_admin import firestore

# Gemini sozlamalari (Sizning API kilitingiz)
genai.configure(api_key="SIZNING_GEMINI_API_KILITINGIZ")
db = firestore.client()

async def get_smart_answer_for_bot(user_question: str) -> str:
    try:
        # 1-Qadam: Bot foydalanuvchi so'rovini xuddi extension kabi vektorga o'giradi
        embedding_result = genai.embed_content(
            model="models/text-embedding-004", # Extension qaysi modelni ishlatsa shuni yozing
            content=user_question,
            task_type="retrieval_query"
        )
        query_vector = embedding_result['embedding']

        # 2-Qadam: Firestore-dagi "search_index" kolleksiyasidan eng o'xshash matnlarni qidirish
        collection_ref = db.collection('search_index') # Extension ulangan kolleksiya nomi
        vector_query = collection_ref.find_nearest(
            vector_field="embedding",
            query_vector=query_vector,
            distance_measure=DistanceMeasure.COSINE,
            limit=3  # Faqat eng yaxshi 3 ta natijani olamiz
        )
        
        results = vector_query.get()
        
        # 3-Qadam: Topilgan ma'lumotlarni bitta matnga yig'ish (Context)
        context_texts = []
        for doc in results:
            doc_data = doc.to_dict()
            # Extension 'input' degan maydondan vektor yasayotgan edi, shuni o'qiymiz
            context_texts.append(doc_data.get('input', '')) 
            
        full_context = "\n\n---\n\n".join(context_texts)

        # 4-Qadam: Gemini'ga faqat shu topilgan yo'riqnomalar asosida javob yozdiramiz
        prompt = f"""
        Siz "Sud Qo'llanma" tizimining aqlli yordamchisisiz.
        Foydalanuvchiga faqat quyidagi ma'lumotlar asosida javob bering.
        Agar ma'lumotlar ichida javob bo'lmasa, "Kechirasiz, bu haqida yo'riqnoma topilmadi" deb ayting.
        
        Ma'lumotlar:
        {full_context}
        
        Foydalanuvchi savoli: {user_question}
        """
        
        # Generativ model orqali inson tilida javob qaytarish
        generative_model = genai.GenerativeModel('gemini-1.5-pro') # Yoki gemini-1.5-flash
        response = generative_model.generate_content(prompt)
        
        return response.text

    except Exception as e:
        print(f"Vektor qidiruvida xatolik: {e}")
        return "Tizimda xatolik yuz berdi, iltimos keyinroq qayta urinib ko'ring."
